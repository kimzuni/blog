## Bash
```bash
read t
read n
while [ 0 -lt $n ]; do
	read p m
	t=$((t - p*m))
	n=$((n - 1))
done
[ $t == 0 ] && echo "Yes" || echo "No"
```

## C
```c
#include <stdio.h>

int main(void) {
	int t, n, p, m;
	scanf("%d", &t);
	scanf("%d", &n);
	while (0 < n) {
		scanf("%d %d", &p, &m);
		t-=p*m;
		n--;
	}
	printf("%s\n", t == 0 ? "Yes" : "No");
	return 0;
}
```

## Node.js

## Python3
```python
t = int(input())
n = int(input())
for i in range(n):
    p, m = map(int, input().split())
    t-=p*m
print("Yes" if t==0 else "No")
```

## Ruby
```ruby
t = gets.chomp.to_i
n = gets.chomp.to_i
for i in 0...n
  p, m = gets.chomp.split().map {|i| i.to_i}
  t-=p*m
end
puts t == 0 ? "Yes" : "No"
```
