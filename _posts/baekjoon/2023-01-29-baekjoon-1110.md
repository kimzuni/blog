## Bash
```bash
read n
m=$n
i=1
while :; do
	a=$((m%10))
	b=$((m/10 + a))
	m=$((a*10 + b%10))
	if [ $m == $n ]; then
		break
	fi
	i=$((i + 1))
done
echo $i
```

## C
```c
#include <stdio.h>

int main(void) {
	int n;
	scanf("%d", &n);

	int m = n, i = 0;
	do {
		int a = m%10;
		int b = m/10 + a;
		m = a*10 + b%10;
		i++;
	} while (n != m);
	printf("%d\n", i);

	return 0;
}
```

## Node.js
