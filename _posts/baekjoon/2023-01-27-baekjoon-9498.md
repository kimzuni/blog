## Bash
```bash
read a
o="F"
if [ 90 -le $a ]; then o="A"
elif [ 80 -le $a ]; then o="B"
elif [ 70 -le $a ]; then o="C"
elif [ 60 -le $a ]; then o="D"
fi
echo $o
```

## C
```c
#include <stdio.h>

int main(void) {
	int a;
	scanf("%d", &a);

	if (90 <= a) { printf("A\n"); }
	else if (80 <= a) { printf("B\n"); }
	else if (70 <= a) { printf("C\n"); }
	else if (60 <= a) { printf("D\n"); }
	else { printf("F\n"); }

	return 0;
}
```
