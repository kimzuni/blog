## Bash
```bash
read n
i=$n
while [ 0 -lt $i ]; do
	read oxs
	j=0
	bak=0
	sum=0
	while [ $j -lt ${#oxs} ]; do
		[ "${oxs:$j:1}" == "O" ] && g=$((bak + 1)) || g=0
		sum=$((sum + g))
		bak=$g
		j=$((j + 1))
	done
	echo $sum

	i=$((i - 1))
done
```

## C
```c
#include <stdio.h>
#include <string.h>

int main(void) {
	int n, g, sum, bak;
	scanf("%d", &n);

	for (int i=0; i<n; i++) {
		sum = 0;
		bak = 0;
		char input[n+1];
		scanf("%s", input);

		for (int j=0; j<strlen(input); j++) {
			g = input[j] == 'O' ? bak + 1 : 0;
			sum += g;
			bak = g;
		}

		printf("%d\n", sum);
	}

	return 0;
}
```

## Python3
```python
n = int(input())
for i in range(n):
    oxs = input()
    arr = [0]
    for ox in oxs:
        g = arr[-1] + 1 if ox == "O" else 0
        arr.append(g)
    print(sum(arr))
```
