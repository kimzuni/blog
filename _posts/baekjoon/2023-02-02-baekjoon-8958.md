한 게임당 얻을 수 있는 최대 점수는?

## Bash
```bash
read n
for ((i=0; i<n; i++)); do
	read oxs
	g=0
	sum=0
	for ((j=0; j<${#oxs}; j++)); do
		[ "${oxs:$j:1}" == "O" ] && g=$((g + 1)) || g=0
		sum=$((sum + g))
	done
	echo $sum
done
```

## C
```c
#include <stdio.h>
#include <string.h>

int main(void) {
	int n, g, sum;
	scanf("%d", &n);

	for (int i=0; i<n; i++) {
		sum = 0;
		char input[n+1];
		scanf("%s", input);

		for (int j=0; j<strlen(input); j++) {
			g = input[j] == 'O' ? g + 1 : 0;
			sum += g;
		}

		printf("%d\n", sum);
	}

	return 0;
}
```

## Python3
```python
n = int(input())
for i in range(n):
    oxs = input()
    arr = [0]
    for ox in oxs:
        g = arr[-1] + 1 if ox == "O" else 0
        arr.append(g)
    print(sum(arr))
```
