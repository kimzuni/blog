## Bash
```bash
read a b
read c
b=$((b + c))
a=$((a + b/60))
a=$((a%24))
b=$((b%60))
echo $a $b
```

## C
```c
#include <stdio.h>

int main(void) {
	int a, b, c;
	scanf("%d %d", &a, &b);
	scanf("%d", &c);

	b += c;
	a += b/60;
	a = a%24;
	b = b%60;
	printf("%d %d", a, b);

	return 0;
}
```

## Node.js
```javascript
let [ab, c] = require("fs").readFileSync(0).toString().trim().split("\n");
```

## Python3

## Ruby
