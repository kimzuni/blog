## Bash
```bash
read a
if [ $(($a%4)) == 0 -a $(($a%100)) != 0 ] || [ $(($a%400)) == 0 ]; then o=1
else o=0
fi
echo $o
```

## C
```c
#include <stdio.h>

int main(void) {
	int a;
	scanf("%d", &a);
	if ( (a%4 == 0 && a%100 != 0) || a%400 == 0 ) { printf("1"); }
	else { printf("0"); }
}
```
