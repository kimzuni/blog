## Bash
```bash
read a b
o="=="
if [ $a -lt $b ]; then
	o="<"
elif [ $a -gt $b ]; then
	o=">"
fi
echo $o
```

## C
```c
#include <stdio.h>

int main(void) {
	int a, b;
	scanf("%d %d", &a, &b);

	if (a < b) {
		printf("<\n");
	} else if (a > b) {
		printf(">\n");
	} else {
		printf("==\n");
	}

	return 0;
}
```

## Node.js
```javascript
let input = require("fs").readFileSync("/dev/stdin").toString().split(" ").map(x => parseInt(x));
let a = input[0];
let b = input[1];
let o = "==";
if (a < b) {
	o = "<";
} else if (a > b) {
	o = ">";
}
console.log(o);
```
