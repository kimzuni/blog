## Bash
```bash
arr=()
for i in {1..28}; do
	read n
	arr=(${arr[@]} $n)
done
for i in {1..30}; do
	if ! [[ " ${arr[@]} " =~ " $i " ]]; then
		echo $i
	fi
done
```

## C
```c
#include <stdio.h>

int main(void) {
	int arr[28];
	for (int i=0; i<28; i++) {
		scanf("%d", &arr[i]);
	}

	int idx;
	for (int n=1; n<=30; n++) {
		idx = -1;
		for (int i=0; i<28; i++) {
			if (arr[i] == n) {
				idx = i;
				break;
			}
		}
		if (idx == -1) {
			printf("%d\n", n);
		}
	}

	return 0;
}
```

## Node.js
```javascript
let arr = [...Array(30).keys()].map(x => x + 1)
let input = require("fs").readFileSync(0).toString().trim().split("\n").map(Number).sort();
for (let n of input) {
	arr.splice(arr.indexOf(n), 1);
}
console.log(arr.join("\n"))
```
