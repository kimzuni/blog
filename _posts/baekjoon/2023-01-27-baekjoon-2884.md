## Bash
```bash
read a b
b=$((b - 45))
if [ $b -lt 0 ]; then
	a=$(((a-1+24)%24))
	b=$(((b+60)%60))
fi
echo $a $b
```

## C
```c
#include <stdio.h>

int main(void) {
	int a, b;
	scanf("%d %d", &a, &b);

	b -= 45;
	if ( b < 0 ) {
		a = (a-1+24)%24;
		b = (b+60)%60;
	}
	printf("%d %d", a, b);

	return 0;
}
```
