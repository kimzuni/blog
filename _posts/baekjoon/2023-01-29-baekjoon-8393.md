## Bash
```bash
read n
s=0
while [ 0 -lt $n ]; do
	s=$((s + n))
	n=$((n - 1))
done
echo $s
```

## C
```c
#include <stdio.h>

int main(void) {
	int n, s = 0;
	scanf("%d", &n);
	for (int i=1; i<=n; i++) {
		s+=i;
	}
	printf("%d\n", s);
	return 0;
}
```

## Node.js
```javascript
let n = Number(require("fs").readFileSync(0).toString().trim().split("\n")[0]);
let s = 0;
while (0 < n) {
	s += n;
	n--;
}
console.log(s);
```
