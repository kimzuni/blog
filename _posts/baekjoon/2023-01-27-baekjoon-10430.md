## Bash
```bash
read a b c
echo $((($a + $b) % c))
echo $(((($a % $c) + ($b % $c)) % c))
echo $((($a * $b) % c))
echo $(((($a % $c) * ($b % $c)) % c))
```

## C
```c
#include <stdio.h>

int main(void) {
	int a, b, c;
	scanf("%d %d %d", &a, &b, &c);
	printf("%d\n", (a+b)%c);
	printf("%d\n", ((a%c)+(b%c))%c);
	printf("%d\n", (a*b)%c);
	printf("%d\n", ((a%c)*(b%c))%c);
	return 0;
}
```

## Node.js
```javascript
let input = require("fs").readFileSync("/dev/stdin").toString().split(" ").map(x => parseInt(x));
let a = input[0];
let b = input[1];
let c = input[2];
console.log((a+b)%c);
console.log(((a%c)+(b%c))%c);
console.log((a*b)%c);
console.log(((a%c)*(b%c))%c);
```
