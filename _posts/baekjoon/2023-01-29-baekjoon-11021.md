## Bash
```bash
read n
i=1
while [ $i -le $n ]; do
	read a b
	echo Case \#$i: $((a + b))
	i=$((i + 1))
done
```

## C
```c
#include <stdio.h>

int main(void) {
	int n, a, b;
	scanf("%d", &n);
	for (int i=0; i<n; i++) {
		scanf("%d %d", &a, &b);
		printf("Case #%d: %d\n", i+1, a+b);
	}
	return 0;
}
```

## Node.js
```javascript
let [n, ...list] = require("fs").readFileSync(0).toString().trim().split("\n");

	console.log(`Case #${i*1+1}: ${a+b}`);
}
```

## Python3
```python
n = int(input())
for i in range(n):
    a, b = map(int, input().split())
    print("Case #{}: {}".format(i+1, a+b))
```
