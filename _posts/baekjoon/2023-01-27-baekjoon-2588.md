## Bash
```bash
read a
read b
b3=${b::1}
b2=${b:1:1}
b1=${b:2}
echo $(($a * $b1))
echo $(($a * $b2))
echo $(($a * $b3))
echo $(($a * $b))
```

## C
```c
#include <stdio.h>

int main(void) {
	int a, b;
	scanf("%d", &a);
	scanf("%d", &b);

	int c1, c2, c3;
	c3 = b/100;
	c2 = (b-c3*100)/10;
	c1 = (b-c3*100)%10;

	printf("%d\n", a*c1);
	printf("%d\n", a*c2);
	printf("%d\n", a*c3);
	printf("%d\n", a*b);

	return 0;
}
```

## Node.js
